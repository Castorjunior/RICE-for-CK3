activity_siwa_spring_visit = {
	solo = yes
	frame = 1
	color = { 0.05 0.88 0.96 1 } #???
	on_spawn = {
		invite_character_to_activity = activity_owner
		accept_invitation_for_character = activity_owner

	}
	on_activate = {
		activity_owner = {
			trigger_event = {
				id = siwa.0011
				days = { 4 12 }
			}
		}
	}
	on_complete = {
		activity_owner = {
			remove_character_flag = planning_an_activity
		}	
	}
}
